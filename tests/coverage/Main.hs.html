    8 module Main where
    9 import System.Exit (exitWith, ExitCode(..))
   10 import System.Environment (getArgs)
   11 import Data.Char (isDigit)
   12 import CommandChecker (isSorted, doOperation)
   13 import Text.Read (readMaybe)
   14 
   15 isOperator :: String -> Bool
   16 isOperator "sa" = True
   17 isOperator "sb" = True
   18 isOperator "sc" = True
   19 isOperator "pa" = True
   20 isOperator "pb" = True
   21 isOperator "ra" = True
   22 isOperator "rb" = True
   23 isOperator "rr" = True
   24 isOperator "rra" = True
   25 isOperator "rrb" = True
   26 isOperator "rrr" = True
   27 isOperator " " = False
   28 isOperator _ = False
   29 
   30 parseArgs :: String -> Maybe [String]
   31 parseArgs [] = Just []
   32 parseArgs str
   33     | length str >= 2 && isOperator (take 2 str) =
   34         case parseArgs (drop 2 str) of
   35             Just ops -> Just (take 2 str : ops)
   36             Nothing -> Just [take 2 str]
   37     | otherwise = case head str of
   38         ' ' -> parseArgs (tail str)
   39         _ -> Just ["IP"]
   40 
   41 myReadMaybe :: String -> Int
   42 myReadMaybe s = case readMaybe s of
   43     Just x -> x
   44     Nothing -> -1
   45 
   46 parseInts :: [String] -> [Int]
   47 parseInts [] = []
   48 parseInts (x:xs)
   49     | all isDigit x = myReadMaybe x : parseInts xs
   50     | otherwise = (-1) : parseInts xs
   51 
   52 hasInvalidOp :: [String] -> Bool
   53 hasInvalidOp [] = False
   54 hasInvalidOp ["IP"] = True
   55 hasInvalidOp (_:xs) = hasInvalidOp xs
   56 
   57 hasInvalidInt :: [Int] -> Bool
   58 hasInvalidInt [] = False
   59 hasInvalidInt (x:xs)
   60     | x == -1 = True
   61     | otherwise = hasInvalidInt xs
   62 
   63 main :: IO ()
   64 main = do
   65     args <- getArgs
   66     let i = parseInts args
   67     if null args || hasInvalidInt i then
   68         exitWith (ExitFailure 84)
   69     else do
   70         line <- getLine
   71         let s = parseArgs line
   72         processArgs i s
   73 
   74 processArgs :: [Int] -> Maybe [String] -> IO ()
   75 processArgs l_a (Just args)
   76     | null args = exitWith (ExitFailure 84)
   77     | hasInvalidOp args = exitWith (ExitFailure 84)
   78     | otherwise =
   79         let (final_l_a, final_l_b) = foldl doOperation (l_a, []) args
   80         in putStrLn $ resultMessage final_l_a final_l_b
   81 processArgs _ Nothing = exitWith (ExitFailure 84)
   82 
   83 resultMessage :: [Int] -> [Int] -> String
   84 resultMessage final_l_a final_l_b
   85     | isSorted final_l_a = "OK"
   86     | otherwise = "KO: (" ++ show final_l_a ++ "," ++ show final_l_b ++ ")"
