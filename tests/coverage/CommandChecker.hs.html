    8 module CommandChecker (isSorted, doOperation) where
    9 import System.Environment()
   10 import Data.List (sort)
   11 import Utilities (swap, swapb, pa, pb, rotate, rr, rotaterev, rrr)
   12 
   13 isSorted :: [Int] -> Bool
   14 isSorted xs = xs == sort xs
   15 
   16 doOperation :: ([Int], [Int]) -> String -> ([Int], [Int])
   17 doOperation (l_a, l_b) "sa" = (swap l_a, l_b)
   18 doOperation (l_a, l_b) "sb" = (l_a, swap l_b)
   19 doOperation (l_a, l_b) "sc" = swapb l_a l_b
   20 doOperation (l_a, l_b) "pa" = pa l_a l_b
   21 doOperation (l_a, l_b) "pb" = pb l_a l_b
   22 doOperation (l_a, l_b) "ra" = (rotate l_a, l_b)
   23 doOperation (l_a, l_b) "rb" = (l_a, rotate l_b)
   24 doOperation (l_a, l_b) "rr" = rr l_a l_b
   25 doOperation (l_a, l_b) "rra" = (rotaterev l_a, l_b)
   26 doOperation (l_a, l_b) "rrb" = (l_a, rotaterev l_b)
   27 doOperation (l_a, l_b) "rrr" = rrr l_a l_b
   28 doOperation (l_a, l_b) _ = (l_a, l_b)
