    8 module Utilities (swap, swapb, pa, pb, rotate, rr, rotaterev, rrr) where
    9 
   10 swap :: [a] -> [a]
   11 swap [] = []
   12 swap [x] = [x]
   13 swap (x:y:xs) = y : x : xs
   14 
   15 swapb :: [a] -> [a] -> ([a], [a])
   16 swapb l1 l2 = (swap l1, swap l2)
   17 
   18 pa :: [a] -> [a] -> ([a], [a])
   19 pa l1 [] = (l1, [])
   20 pa l1 (x:xs) = (x : l1, xs)
   21 
   22 pb :: [a] -> [a] -> ([a], [a])
   23 pb [] l2 = ([], l2)
   24 pb (x:xs) l2 = (xs, x : l2)
   25 
   26 rotate :: [a] -> [a]
   27 rotate [] = []
   28 rotate (x:xs) = xs ++ [x]
   29 
   30 rr :: [a] -> [a] -> ([a], [a])
   31 rr l1 l2 = (rotate l1, rotate l2)
   32 
   33 rotaterev :: [a] -> [a]
   34 rotaterev [] = []
   35 rotaterev list = last list : init list
   36 
   37 rrr :: [a] -> [a] -> ([a], [a])
   38 rrr l1 l2 = (rotaterev l1, rotaterev l2)
